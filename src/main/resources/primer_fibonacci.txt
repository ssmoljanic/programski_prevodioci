# Racuna prvih N Fibonacci brojeva

recept porudzbina fibonacci(porudzbina n) {
    rezervisanSto(n <= 1) {
        return n;
    }
    return fibonacci(n manje 1) ukupno fibonacci(n manje 2);
}
