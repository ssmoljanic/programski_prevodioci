recept stampajMatricu(listaCekanja{porudzbina}[2][3] mat){
    novaTura(izvolite porudzbina i je 0; i < 2; i je i ukupno 1){
        novaTura(izvolite porudzbina j je 0; j < 3; j je j ukupno 1){
            konobar(mat[i][j]);
        }
    }
    return;
}

        {1,2,3},
        {4,5,6}
    };

    izvolite porudzbina suma je 0;

    novaTura(izvolite porudzbina i je 0; i < 2; i je i ukupno 1){
        novaTura(izvolite porudzbina j je 0; j < 3; j je j ukupno 1){

}
